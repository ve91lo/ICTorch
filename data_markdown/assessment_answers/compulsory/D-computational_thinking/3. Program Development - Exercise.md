# Bank Account Management

```python
balance = 0

def check_balance():
print(f"Your current balance is ${balance:.2f}")

def deposit(amount):
global balance
if amount > 0:
balance += amount
print(f"You have deposited ${amount:.2f}. New balance is ${balance:.2f}")
else:
print("Deposit amount must be positive.")

def withdraw(amount):
global balance
if 0 &#x3C; amount &#x3C;= balance:
balance -= amount
print(f"You have withdrawn ${amount:.2f}. New balance is ${balance:.2f}")
elif amount > balance:
print("Insufficient funds.")
else:
print("Withdrawal amount must be positive.")

# Example usage
check_balance()
deposit(100.50)
check_balance()
withdraw(30)
check_balance()
withdraw(80)
check_balance()
